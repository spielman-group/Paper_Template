% command declarations for reviewer points and our responses
\newcommand{\reviewersection}{\stepcounter{reviewer} \bigskip \hrule
    \section*{Reviewer \thereviewer : Specific Comments and Critiques}}

% Reviewer's point with subpoint numbering
\newenvironment{point}[1]
{
    \bigskip \refstepcounter{point} \label{#1}
    \noindent {\large\textbf{Reviewer~Point~\thepoint}} \bigskip
    \begin{quotebox}{\refstepcounter{subpoint} \textbf{\thesubpoint} --- }
}
{
    \end{quotebox}
}

% Reviewer's point without subpoint numbering
\newenvironment{point*}[1]
{
    \bigskip \refstepcounter{point} \label{#1}
    \noindent {\large\textbf{Reviewer~Point~\thepoint}}
    \bigskip
    \begin{quotebox}{}
}
{
    \end{quotebox}
}

\newenvironment{subpoint}[1]
{
    \refstepcounter{subpoint} \label{#1}
    \begin{quotebox}{{\textbf{\thesubpoint} --- }}
}
{
    \end{quotebox}
}
